CREATE TABLE `member`(
	id INT UNSIGNED NOT NULL PRIMARY KEY AUTO_INCREMENT,
	regDate DATETIME NOT NULL,
	updateDate DATETIME NOT NULL,
	loginId VARCHAR(20) NOT NULL UNIQUE,
	loginPw VARCHAR(100) NOT NULL,
	company VARCHAR(20) NOT NULL,
	`name` VARCHAR(20) NOT NULL,
	nickname VARCHAR(20) NOT NULL,
	cellphoneNum VARCHAR(20) NOT NULL,
	email VARCHAR(50) NOT NULL,
	`position` INT UNSIGNED NOT NULL DEFAULT 0 COMMENT '부장 이상 0, 부장 이하 1',
	depart VARCHAR(20) NOT NULL,
	pw VARCHAR(20)
	
);
CREATE TABLE notice(
	 id INT UNSIGNED NOT NULL PRIMARY KEY AUTO_INCREMENT,
   regDate DATETIME NOT NULL,
   updateDate DATETIME NOT NULL,
   title VARCHAR(100) NOT NULL,
   `body` TEXT NOT NULL,
   memberId SMALLINT UNSIGNED NOT NULL,
   company VARCHAR(20) NOT NULL
)
CREATE TABLE project (
	 id INT UNSIGNED NOT NULL PRIMARY KEY AUTO_INCREMENT,
   startDate DATETIME NOT NULL,
   endDate DATETIME NOT NULL,
   title VARCHAR(100) NOT NULL,
   `body` TEXT NOT NULL,
   manager INT UNSIGNED NOT NULL,
   FOREIGN KEY (manager) REFERENCES `member`(id)   
)
CREATE TABLE project_member (
   regDate DATETIME NOT NULL,
   updateDate DATETIME,
   project_id INT UNSIGNED NOT NULL,
   member_id INT UNSIGNED NOT NULL,
   PRIMARY KEY (project_id, member_id),
   FOREIGN KEY (project_id) REFERENCES project(id),
   FOREIGN KEY (member_id) REFERENCES `member`(id)
);
CREATE TABLE task (
	 id INT UNSIGNED NOT NULL PRIMARY KEY AUTO_INCREMENT,
	 member_id INT UNSIGNED NOT NULL,
	 project_id INT UNSIGNED NOT NULL,
   startDate DATETIME NOT NULL,
   endDate DATETIME NOT NULL,
   title VARCHAR(100) NOT NULL,
   `body` TEXT NOT NULL,
   `status` SMALLINT UNSIGNED NOT NULL DEFAULT 1 COMMENT '작업 생성 1, 진행 중 2, 완료 3',
   FOREIGN KEY (member_id) REFERENCES `member`(id),
   FOREIGN KEY (project_id) REFERENCES project(id)
      
)
